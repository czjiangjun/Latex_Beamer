	\begin{displaymath}
		\Psi=C\left\{ \mathrm{exp}[－Z(r_1+ p_2)/a]+\mathrm{exp}[－Z( r_2+ p_1)/a] \right\}
	\end{displaymath}}
	{\fontsize{7.2pt}{6.5pt}\selectfont{其中$r_1$、$p_1$是第一个电子到两个原子核的距离，$r_2$，$p_2$是第二个电子到两个原子核的距离，$a$是\textrm{Bohr}半径}}\\

	\textcolor{blue}{
	\begin{displaymath}
		E= (hc8\pi)[Aj(j+1)+W]
	\end{displaymath}}
	\end{itemize}
